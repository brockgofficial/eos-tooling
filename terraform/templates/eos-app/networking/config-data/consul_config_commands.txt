## Commands to add config data to our Consul installation ##

# Launch consul server instance
consul agent -bootstrap -config-file="config/consul-config.hcl" -bind="127.0.0.1"

## Open a separate terminal window to run the rest of the commands
# Set the Consul token

# Windows
$env:CONSUL_HTTP_TOKEN="VALUE"

# Write the configuration data to consul

# Change to directory of configuration data "eos-tooling\terraform\templates\eos-app\networking\config-data"

consul kv put networking/configuration/net_info @net-config.json
consul kv put networking/configuration/common_tags @common-tags.json
